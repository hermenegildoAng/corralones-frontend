<script setup>
import { ref, computed, onMounted } from 'vue'
import { useRouter } from 'vue-router'
import { useAuthStore } from '../stores/auth'
import clienteAxios from '../api/axios'
import { PlusIcon, EyeIcon, MagnifyingGlassIcon } from '@heroicons/vue/24/outline'

const auth = useAuthStore()
const router = useRouter()

// --- ESTADOS ---
const sedes = ref([])        // Esta es nuestra lista maestra de corralones
const search = ref('')       // Buscador de texto
const filtroEstatus = ref('TODOS')
const loading = ref(false)
const mostrarModal = ref(false)

// Filtros de Rango (Ocupación)
const filtroMin = ref(null)
const filtroMax = ref(null)

const nuevoDep = ref({
  nombre: '',
  ubicacion: '',
  concesionario_responsable: '',
  estatus_deposito: 'ACTIVO'
})

// --- CARGAR DATOS ---
const cargarDatos = async () => {
  try {
    // Solo pedimos depósitos porque el Serializer ya trae el conteo de unidades
    const res = await clienteAxios.get('depositos/')
    sedes.value = res.data
  } catch (e) {
    console.error("Error cargando depósitos:", e)
  }
}

// --- LÓGICA DE FILTRADO ---
const filteredDepositos = computed(() => {
  return sedes.value.filter(s => {
    const query = search.value.toLowerCase()
    
    // 1. Filtro por nombre o concesionario (Blindado contra nulos)
    const matchesSearch = 
      (s.nombre?.toLowerCase() || '').includes(query) || 
      (s.concesionario_responsable?.toLowerCase() || '').includes(query)

    // 2. Filtro por Estatus (ACTIVO/SUSPENDIDO)
    const matchesEstatus = filtroEstatus.value === 'TODOS' || s.estatus_deposito === filtroEstatus.value

    // 3. Filtro por Rango de Ocupación (Usando total_unidades del Backend)
    const unidades = s.total_unidades || 0
    const matchesMin = filtroMin.value === null || filtroMin.value === '' || unidades >= filtroMin.value
    const matchesMax = filtroMax.value === null || filtroMax.value === '' || unidades <= filtroMax.value

    return matchesSearch && matchesEstatus && matchesMin && matchesMax
  })
})

// --- ACCIONES ---
const crearDeposito = async () => {
  loading.value = true
  try {
    await clienteAxios.post('depositos/', nuevoDep.value)
    mostrarModal.value = false
    nuevoDep.value = { nombre: '', ubicacion: '', concesionario_responsable: '', estatus_deposito: 'ACTIVO' }
    await cargarDatos() // Refrescamos la lista
  } catch (e) {
    alert('Error al registrar el depósito.')
  } finally {
    loading.value = false
  }
}

const limpiarRango = () => {
  filtroMin.value = null
  filtroMax.value = null
}

const verDetalles = (id) => {
  localStorage.setItem('deposito_ver_id', id)
  router.push('/DetallesDeposito')
}

const claseEstatus = (estatus) => {
  switch(estatus) {
    case 'ACTIVO': return 'bg-green-50 text-green-600 border-green-100'
    case 'SUSPENDIDO': return 'bg-red-50 text-red-600 border-red-100'
    default: return 'bg-amber-50 text-amber-600 border-amber-100'
  }
}

onMounted(cargarDatos)
</script>
