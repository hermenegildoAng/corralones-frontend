<template>
  <div class="flex min-h-screen bg-white font-sans relative overflow-x-hidden">
    
    <transition name="drawer-left">
      <aside v-if="drawerSolicitud" class="fixed left-0 top-0 h-screen w-[450px] bg-white border-r border-slate-200 shadow-2xl z-[70] flex flex-col p-10 overflow-y-auto">
        <div class="flex justify-between items-start mb-10">
          <div>
            <span class="text-[9px] font-black text-rosa-comp uppercase tracking-[0.4em]">Propuesta de Cambio</span>
            <h2 class="text-2xl font-black text-slate-800 tracking-tighter uppercase mt-1">Corregir Registro</h2>
          </div>
          <button @click="cerrarEditor" class="p-2 hover:bg-slate-50 rounded-xl transition-all">✕</button>
        </div>

        <div class="space-y-8">
          <div class="p-6 bg-slate-50 rounded-[2.5rem] border border-slate-100 shadow-inner">
            <p class="text-[10px] font-black text-slate-400 uppercase tracking-widest mb-2">Campo a Editar</p>
            <p class="text-xs font-black text-slate-800 uppercase">{{ campoAEditar.label }}</p>
          </div>

          <div v-if="!campoAEditar.esFoto" class="space-y-4">
            <label class="text-[10px] font-black text-slate-400 uppercase tracking-widest ml-4">Nuevo Valor Propuesto</label>
            <input 
              v-if="campoAEditar.tipo === 'text'"
              v-model="formEdicion.valor_nuevo" 
              type="text" 
              class="w-full bg-slate-50 border-2 border-transparent rounded-2xl p-5 text-xs font-black uppercase focus:border-rosa-comp/20 outline-none transition-all text-slate-700"
              placeholder="Escriba el dato correcto"
            />
            <select 
              v-else-if="campoAEditar.tipo === 'select'"
              v-model="formEdicion.valor_nuevo"
              class="w-full bg-slate-50 border-none rounded-2xl p-5 text-xs font-black uppercase outline-none text-slate-700"
            >
              <option v-for="opt in campoAEditar.opciones" :key="opt" :value="opt">{{ opt }}</option>
            </select>
          </div>

          <div class="space-y-4">
            <label class="text-[10px] font-black text-slate-400 uppercase tracking-widest ml-4">Prueba del error (Foto)</label>
            <div class="border-2 border-dashed border-slate-100 rounded-[2.5rem] p-8 text-center hover:bg-slate-50 transition-all cursor-pointer relative overflow-hidden group">
              <input type="file" class="absolute inset-0 opacity-0 cursor-pointer z-10" @change="handleEvidencia" accept="image/*" />
              <div v-if="!evidenciaPreview" class="space-y-2">
                <PhotoIcon class="w-10 h-10 text-slate-200 mx-auto" />
                <p class="text-[9px] font-black text-slate-300 uppercase">Click para subir foto</p>
              </div>
              <img v-else :src="evidenciaPreview" class="w-full h-48 object-cover rounded-2xl shadow-lg" />
            </div>
          </div>

          <div class="space-y-4">
            <label class="text-[10px] font-black text-slate-400 uppercase tracking-widest ml-4">Justificación Técnica</label>
            <textarea v-model="formEdicion.justificacion" rows="4" class="w-full bg-slate-50 border-none rounded-[2rem] p-6 text-sm outline-none italic resize-none text-slate-600" placeholder="Explique el porqué del cambio"></textarea>
          </div>

          <button @click="enviarSolicitud" class="w-full py-6 bg-primario text-white rounded-[2rem] text-[10px] font-black uppercase tracking-[0.3em] shadow-xl active:scale-95 transition-all">
            Enviar Solicitud
          </button>
        </div>
      </aside>
    </transition>

    <div :class="['transition-all duration-500 flex-1 p-4 md:p-10 space-y-10', drawerSolicitud ? 'ml-[450px]' : '', drawerInspeccion ? 'mr-[400px]' : '']">
      <div v-if="!loading" class="max-w-7xl mx-auto">
        
        <header class="flex flex-col md:flex-row justify-between items-start md:items-end mb-10 gap-6 border-b border-slate-100 pb-8">
          <div>
            <span class="text-[10px] font-black text-rosa-comp uppercase tracking-[0.3em]">Expediente Maestro</span>
            <h1 class="text-4xl font-black text-slate-800 tracking-tighter uppercase mt-2">
              {{ vehiculo.marca }} {{ vehiculo.modelo }} <span class="text-slate-300 font-light">{{ vehiculo.anio }}</span>
            </h1>
            <p class="text-slate-400 font-medium text-sm mt-1 italic">Folio: <span class="text-primario font-black">{{ ingreso.folio }}</span></p>
          </div>

          <div class="flex items-center gap-6 bg-white p-6 rounded-[2.5rem] shadow-xl border border-slate-100">
            <div v-if="auth.rol !== 'ADMIN'" class="flex items-center gap-4 pr-6 border-r border-slate-100">
              <span class="text-[9px] font-black text-slate-400 uppercase tracking-widest italic">Modo Edición</span>
              <button @click="modoEdicion = !modoEdicion" :class="['w-14 h-7 rounded-full relative transition-all duration-500', modoEdicion ? 'bg-rosa-comp shadow-lg shadow-rosa-comp/30' : 'bg-slate-200']">
                <div :class="['absolute top-1 w-5 h-5 bg-white rounded-full transition-all shadow-sm', modoEdicion ? 'left-8' : 'left-1']"></div>
              </button>
            </div>
          </div>
        </header>

        <section class="grid grid-cols-1 md:grid-cols-5 gap-4 mb-10">
          <div class="md:col-span-3 aspect-video bg-white rounded-[3rem] p-2 border border-slate-100 shadow-sm relative overflow-hidden group">
            <img :src="fotoPrincipal" class="w-full h-full object-cover rounded-[2.5rem] transition-transform duration-700 group-hover:scale-105" />
            <button v-if="modoEdicion" @click="abrirEditor('Foto Principal', fotoPrincipal, true)" class="absolute top-10 right-10 p-3 bg-white/90 backdrop-blur text-rosa-comp rounded-2xl shadow-xl font-black text-[9px] uppercase border border-rosa-comp/20 z-20">Cambiar Foto</button>
          </div>
          <div class="md:col-span-2 grid grid-cols-2 gap-4">
            <div v-for="(img, key) in galeriaReal" :key="key" @click="fotoPrincipal = img" class="bg-white p-1.5 rounded-3xl border border-slate-100 shadow-sm cursor-pointer hover:ring-2 hover:ring-primario transition-all overflow-hidden relative group">
              <img :src="img" class="w-full h-full object-cover rounded-2xl opacity-80 group-hover:opacity-100" />
              <button v-if="modoEdicion" @click.stop="abrirEditor('Galería: ' + key, img, true)" class="absolute inset-0 bg-rosa-comp/60 opacity-0 group-hover:opacity-100 transition-all flex items-center justify-center text-white text-[9px] font-black uppercase rounded-2xl z-20">Cambiar</button>
            </div>
          </div>
        </section>

        <div class="grid lg:grid-cols-2 gap-8">
          
          <div class="bg-white rounded-[3rem] p-10 border border-slate-100 shadow-sm">
            <h3 class="text-[10px] font-black text-rosa-comp uppercase tracking-[0.3em] mb-8">Información del Trámite</h3>
            <div class="grid grid-cols-2 gap-y-10 gap-x-6">
              <div v-for="(val, label) in infoIngreso" :key="label">
                <p class="text-[9px] font-black text-slate-300 uppercase tracking-widest mb-1">{{ label }}</p>
                <p class="text-xs font-black text-slate-800 uppercase">{{ val || 'N/A' }}</p>
              </div>
            </div>
          </div>

          <div class="bg-white rounded-[3rem] p-10 border border-slate-100 shadow-sm">
            <h3 class="text-[10px] font-black text-primario uppercase tracking-[0.3em] mb-8">Datos de Unidad</h3>
            <div class="grid grid-cols-2 gap-y-10 gap-x-6">
              <div v-for="(val, label) in infoVehiculo" :key="label">
                <p class="text-[9px] font-black text-slate-300 uppercase tracking-widest mb-1">{{ label }}</p>
                <button @click="abrirEditor(label, val)" :disabled="!modoEdicion" :class="['w-full text-left p-3 rounded-xl text-xs font-black uppercase transition-all', modoEdicion ? 'bg-rosa-comp/5 text-rosa-comp border border-dashed border-rosa-comp/30' : 'text-slate-700 cursor-default']">
                  {{ val }}
                </button>
              </div>
            </div>
          </div>

          <div class="lg:col-span-2 bg-white rounded-[3rem] p-10 border border-slate-100 shadow-sm">
            <h3 class="text-[10px] font-black text-slate-400 uppercase tracking-[0.3em] mb-8">Estado Mecánico y Niveles</h3>
            <div class="grid grid-cols-2 md:grid-cols-4 gap-y-10 gap-x-8">
              <div v-for="(val, label) in infoMecanica" :key="label">
                <p class="text-[9px] font-black text-slate-300 uppercase tracking-widest mb-1">{{ label }}</p>
                <button @click="abrirEditor(label, val)" :disabled="!modoEdicion" :class="['w-full text-left p-3 rounded-xl text-xs font-black uppercase transition-all', modoEdicion ? 'bg-rosa-comp/5 text-rosa-comp border border-dashed border-rosa-comp/30' : 'bg-slate-50 text-slate-600 font-bold']">
                  {{ val }}
                </button>
              </div>
            </div>
          </div>

          <div class="lg:col-span-2">
            <div class="flex justify-between items-center mb-6 ml-6 pr-6">
              <h3 class="text-[10px] font-black text-slate-400 uppercase tracking-[0.3em]">Inventario de Objetos</h3>
              <button v-if="modoEdicion" @click="abrirEditor('Añadir Objeto', '')" class="text-[9px] font-black text-rosa-comp uppercase tracking-widest border-b border-rosa-comp/30">+ Reportar nuevo objeto</button>
            </div>
            <div v-if="objetos.length === 0" class="bg-white p-10 rounded-[2.5rem] border border-dashed border-slate-200 text-center"><p class="text-[10px] font-black text-slate-300 uppercase tracking-widest">Sin objetos reportados</p></div>
            <div v-else class="grid grid-cols-1 md:grid-cols-3 gap-6">
              <div v-for="obj in objetos" :key="obj.id" class="bg-white p-6 rounded-[2.5rem] border border-slate-100 shadow-sm flex items-center gap-4 group">
                <div class="w-16 h-16 rounded-2xl bg-slate-50 overflow-hidden relative flex-shrink-0">
                  <img :src="getUrl(obj.foto)" class="w-full h-full object-cover" />
                  <button v-if="modoEdicion" @click="abrirEditor('Foto: ' + obj.descripcion, obj.foto, true)" class="absolute inset-0 bg-rosa-comp/80 opacity-0 group-hover:opacity-100 transition-opacity flex items-center justify-center text-white text-[8px] font-black uppercase">Cambiar</button>
                </div>
                <div class="flex-1">
                  <p class="text-xs font-black text-slate-700 uppercase">{{ obj.descripcion }}</p>
                  <p class="text-[9px] font-bold text-slate-400 uppercase mt-1">{{ obj.cantidad }} PZA • {{ obj.estado_objeto }}</p>
                </div>
                <button v-if="modoEdicion" @click="abrirEditor('Objeto: ' + obj.descripcion, obj.estado_objeto)" class="p-2 text-rosa-comp opacity-0 group-hover:opacity-100 transition-all"><PencilSquareIcon class="w-4 h-4" /></button>
              </div>
            </div>
          </div>

          <div class="lg:col-span-2">
            <div class="flex justify-between items-center mb-6 ml-6 pr-6">
              <h3 class="text-[10px] font-black text-red-400 uppercase tracking-[0.3em]">Daños y Siniestros</h3>
              <button v-if="modoEdicion" @click="abrirEditor('Añadir Daño', '')" class="text-[9px] font-black text-rosa-comp uppercase tracking-widest border-b border-rosa-comp/30">+ Reportar nuevo daño</button>
            </div>
            <div v-if="danos.length === 0" class="bg-white p-10 rounded-[2.5rem] border border-dashed border-slate-200 text-center"><p class="text-[10px] font-black text-slate-300 uppercase tracking-widest">Sin daños físicos reportados</p></div>
            <div v-else class="grid grid-cols-1 md:grid-cols-2 gap-6">
              <div v-for="dano in danos" :key="dano.id" class="bg-white p-6 rounded-[2.5rem] border border-slate-100 shadow-sm flex items-start gap-5 group relative overflow-hidden">
                <div class="absolute left-0 top-0 bottom-0 w-1 bg-red-400"></div>
                <div class="w-24 h-24 rounded-2xl bg-slate-50 overflow-hidden relative flex-shrink-0">
                  <img :src="getUrl(dano.foto)" class="w-full h-full object-cover" />
                  <button v-if="modoEdicion" @click="abrirEditor('Foto Daño: ' + dano.parte_vehiculo, dano.foto, true)" class="absolute inset-0 bg-rosa-comp/80 opacity-0 group-hover:opacity-100 transition-opacity flex items-center justify-center text-white text-[8px] font-black uppercase">Cambiar</button>
                </div>
                <div class="flex-1">
                  <div class="flex justify-between"><p class="text-xs font-black text-red-600 uppercase">{{ dano.parte_vehiculo }}</p>
                    <button v-if="modoEdicion" @click="abrirEditor('Daño: ' + dano.parte_vehiculo, dano.descripcion)" class="p-1 text-primario opacity-0 group-hover:opacity-100 transition-all"><PencilSquareIcon class="w-4 h-4" /></button>
                  </div>
                  <p class="text-[11px] text-slate-600 font-medium italic mt-2">"{{ dano.descripcion }}"</p>
                </div>
              </div>
            </div>
          </div>

          <div class="lg:col-span-2 bg-white rounded-[3rem] p-10 border border-slate-100 shadow-sm">
            <h3 class="text-[10px] font-black text-slate-800 uppercase tracking-[0.3em] mb-8">Historial de Eventos (Bitácora)</h3>
            <div v-if="bitacora.length === 0" class="text-center py-6 text-slate-300 text-xs font-bold uppercase">Sin historial de eventos</div>
            <table v-else class="w-full text-left">
              <thead>
                <tr class="text-[9px] font-black text-slate-300 uppercase border-b border-slate-50">
                  <th class="pb-4">Evento</th>
                  <th class="pb-4">Usuario</th>
                  <th class="pb-4 text-right">Fecha</th>
                </tr>
              </thead>
              <tbody class="divide-y divide-slate-50">
                <tr v-for="log in bitacora" :key="log.id">
                  <td class="py-4">
                    <p class="text-xs font-black text-slate-700 uppercase leading-none">{{ log.tipo_evento }}</p>
                    <p class="text-[9px] text-slate-400 font-bold uppercase mt-1 italic">{{ log.descripcion }}</p>
                  </td>
                  <td class="py-4 text-[10px] font-bold text-slate-400 uppercase italic">@{{ log.user_nombre }}</td>
                  <td class="py-4 text-right text-[10px] font-black text-slate-500">{{ log.fecha_registro }}</td>
                </tr>
              </tbody>
            </table>
          </div>

        </div>
      </div>
      <div v-else class="h-screen flex items-center justify-center">
        <p class="text-sm font-black text-slate-300 animate-pulse uppercase tracking-[0.5em]">Sincronizando Expediente...</p>
      </div>
    </div>
  </div>
</template>

<script setup>
import { ref, computed, onMounted } from 'vue'
import { useAuthStore } from '../stores/auth'
import clienteAxios from '../api/axios'
import { PencilSquareIcon, ClipboardDocumentCheckIcon, DocumentArrowDownIcon, PhotoIcon } from '@heroicons/vue/24/outline'

const auth = useAuthStore()
const API_URL = 'http://127.0.0.1:8000'

// --- ESTADOS ---
const modoEdicion = ref(false)
const drawerSolicitud = ref(false)
const drawerInspeccion = ref(false)
const loading = ref(true)

const ingreso = ref({})
const vehiculo = ref({})
const mecanica = ref({})
const objetos = ref([])
const danos = ref([])
const bitacora = ref([])
const fotoPrincipal = ref('')

const campoAEditar = ref({ label: '', valor: '', tipo: 'text', esFoto: false, opciones: [] })
const formEdicion = ref({ valor_nuevo: '', justificacion: '', evidencia: null })
const evidenciaPreview = ref(null)

const OPCIONES_MAP = {
  'Carrocería': ['BUENO', 'REGULAR', 'MALO', 'DAÑADO'],
  'Motor': ['BUENO', 'REGULAR', 'MALO', 'DAÑADO'],
  'Transmisión': ['MANUAL', 'AUTOMATICA', 'SEMIAUTOMATICA'],
  'Aceite': ['LLENO', 'MEDIO', 'BAJO', 'VACIO'],
  'Combustible': ['LLENO', 'MEDIO', 'BAJO', 'VACIO']
}

const getUrl = (path) => {
  if (!path) return 'https://via.placeholder.com/400x300?text=Sin+Imagen'
  return path.startsWith('http') ? path : `${API_URL}${path}`
}

const cargarExpediente = async () => {
  const id = localStorage.getItem('ingreso_detalle_id')
  try {
    loading.value = true
    const res = await clienteAxios.get(`ingresos/${id}/`)
    const data = res.data
    ingreso.value = data
    vehiculo.value = data.vehiculo || {}
    // Usamos historial_detalles que es el related_name en Django
    mecanica.value = data.historial_detalles || {} 
    objetos.value = data.objetos_personales || []
    danos.value = data.registros_danos || []
    bitacora.value = data.bitacora || []
    fotoPrincipal.value = getUrl(vehiculo.value.foto_frente)
  } catch (e) { console.error(e) } finally { loading.value = false }
}

const galeriaReal = computed(() => ({
  f1: getUrl(vehiculo.value.foto_frente),
  f2: getUrl(vehiculo.value.foto_atras),
  f3: getUrl(vehiculo.value.foto_izq),
  f4: getUrl(vehiculo.value.foto_der)
}))

const infoIngreso = computed(() => ({
  "Folio": ingreso.value.folio,
  "Fecha": ingreso.value.fecha_ingreso,
  "Servicio": ingreso.value.tipo_servicio,
  "Depósito": ingreso.value.nombre_deposito
}))

const infoVehiculo = computed(() => ({
  "Marca": vehiculo.value.marca, "Modelo": vehiculo.value.modelo, "Placas": vehiculo.value.placas, 
  "Serie (VIN)": vehiculo.value.num_serie, "Color": vehiculo.value.color_actual
}))

const infoMecanica = computed(() => ({
  "Carrocería": mecanica.value.estatus_carroceria || 'N/A',
  "Cristales": mecanica.value.estatus_cristales || 'N/A',
  "Espejos": mecanica.value.estatus_espejos || 'N/A',
  "Asientos": mecanica.value.estado_asientos || 'N/A',
  "Motor": mecanica.value.estado_motor || 'N/A',
  "Transmisión": mecanica.value.tipo_transmision || 'N/A',
  "Batería": mecanica.value.presencia_bateria ? 'PRESENTE' : 'AUSENTE',
  "Odómetro": mecanica.value.estatus_odometro || 'N/A',
  "Kilometraje": (mecanica.value.kilometraje_odometro || '0') + ' KM',
  "Aceite Motor": mecanica.value.estatus_aceite_motor || 'N/A',
  "Anticongelante": mecanica.value.estatus_anticongelante || 'N/A',
  "Combustible": mecanica.value.estatus_combustible || 'N/A',
  "Bolsas de Aire": mecanica.value.estatus_bolsas_aire || 'N/A',
  "Llantas Del.": (mecanica.value.cant_llantas_delanteras || '0') + ' Pzas',
  "Llantas Tras.": (mecanica.value.cant_llantas_traseras || '0') + ' Pzas'
}))


const abrirEditor = (label, valor, esFoto = false) => {
  if (!modoEdicion.value) return
  evidenciaPreview.value = null
  campoAEditar.value = { 
    label, valor: esFoto ? '' : valor, esFoto,
    tipo: OPCIONES_MAP[label] ? 'select' : 'text',
    opciones: OPCIONES_MAP[label] || []
  }
  formEdicion.value = { valor_nuevo: esFoto ? '' : valor, justificacion: '', evidencia: null }
  drawerSolicitud.value = true
}

const cerrarEditor = () => { drawerSolicitud.value = false }
const handleEvidencia = (e) => {
  const file = e.target.files[0]
  if (file) {
    evidenciaPreview.value = URL.createObjectURL(file)
    formEdicion.value.evidencia = file
  }
}
const enviarSolicitud = () => { alert('Propuesta enviada'); cerrarEditor() }

onMounted(cargarExpediente)
</script>
