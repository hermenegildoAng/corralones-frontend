<template>
  <div class="p-4 md:p-8 bg-slate-50 min-h-screen flex font-sans">
    
    <!-- CONTENEDOR PRINCIPAL ÚNICO -->
    <div class="bg-white rounded-[2rem] shadow-sm border border-slate-100 w-full flex flex-col xl:flex-row flex-1 overflow-hidden">
      
      <!-- ===== LADO IZQUIERDO: CREDENCIAL / ID DIGITAL ===== -->
      <section class="w-full xl:w-5/12 bg-slate-800 text-white relative flex flex-col">
        <!-- Textura/Patrón superior -->
        <div class="absolute inset-0 opacity-20 bg-[radial-gradient(ellipse_at_top_right,_var(--tw-gradient-stops))] from-primario via-transparent to-transparent"></div>
        
        <div class="p-8 md:p-12 relative z-10 flex flex-col items-center pt-16 flex-1 text-center">
          
          <!-- Badge Estado Activo -->
          <div class="absolute top-8 right-8 flex items-center gap-2 bg-green-500/20 px-4 py-2 rounded-full border border-green-400/30">
            <span class="w-2 h-2 rounded-full bg-green-400 animate-pulse"></span>
            <span class="text-[9px] font-black text-white uppercase tracking-widest">Activo</span>
          </div>

          <div class="w-32 h-32 rounded-full border-4 border-slate-700 bg-slate-900 flex items-center justify-center shadow-2xl overflow-hidden mb-6">
             <UserIcon class="w-16 h-16 text-slate-500" />
          </div>

          <h2 class="text-3xl lg:text-4xl font-black uppercase tracking-tighter">{{ profile.nombre_user || 'Cargando...' }}</h2>
          <p class="text-[10px] sm:text-xs font-black text-primario uppercase tracking-widest mt-2">@{{ profile.username || 'usuario' }}</p>
          
          <div class="mt-8">
            <span class="px-6 py-2.5 bg-white/10 border border-white/20 rounded-full text-[10px] font-black uppercase tracking-[0.2em] shadow-inner">
               {{ profile.rol || 'Rol del Sistema' }}
            </span>
          </div>

          <div class="w-full mt-auto space-y-5 text-left bg-slate-900/50 p-6 md:p-8 rounded-3xl border border-slate-700/50 backdrop-blur-sm">
             <div>
               <p class="text-[8px] font-black text-slate-400 uppercase tracking-widest mb-2">Afiliación Activa</p>
               <div class="flex items-center gap-3">
                 <BuildingOfficeIcon class="w-5 h-5 text-primario" />
                 <p class="text-xs sm:text-sm font-bold text-slate-200 uppercase">{{ profile.corralon || 'Sede SMyT Central' }}</p>
               </div>
             </div>
             <div>
               <p class="text-[8px] font-black text-slate-400 uppercase tracking-widest mb-2">Correo Registrado</p>
               <div class="flex items-center gap-3">
                 <EnvelopeIcon class="w-5 h-5 text-slate-400" />
                 <p class="text-xs sm:text-sm font-bold text-slate-200 truncate">{{ profile.email || 'correo@smyt.gob.mx' }}</p>
               </div>
             </div>
             <div>
               <p class="text-[8px] font-black text-slate-400 uppercase tracking-widest mb-2">Alta en SMyT</p>
               <div class="flex items-center gap-3">
                 <CalendarDaysIcon class="w-5 h-5 text-slate-400" />
                 <p class="text-xs sm:text-sm font-bold text-slate-200 uppercase">{{ profile.fecha_registro || '10 MAR 2026' }}</p>
               </div>
             </div>
          </div>
        </div>
      </section>

      <!-- ===== LADO DERECHO: EDICIÓN Y SEGURIDAD ===== -->
      <section class="w-full xl:w-7/12 p-8 md:p-14 lg:p-20 flex flex-col bg-white">
        
        <div class="flex items-center gap-5 mb-10 border-b border-slate-100 pb-10">
          <div class="p-4 bg-primario/10 rounded-3xl">
            <ShieldCheckIcon class="w-8 h-8 md:w-10 md:h-10 text-primario" />
          </div>
          <div>
            <h3 class="text-3xl md:text-4xl font-black text-slate-800 tracking-tighter uppercase">Seguridad</h3>
            <p class="text-slate-400 text-xs font-bold uppercase tracking-widest mt-1.5">Configura tus credenciales operativas</p>
          </div>
        </div>

        <div v-if="errorMsg" class="mb-10 p-5 bg-red-50 border border-red-100 rounded-3xl text-red-600 text-[10px] sm:text-xs font-black uppercase tracking-widest flex items-center gap-4 shadow-inner">
          <ExclamationTriangleIcon class="w-8 h-8" /> {{ errorMsg }}
        </div>

        <div v-if="loading" class="py-20 text-center text-slate-400 flex flex-col items-center justify-center">
          <div class="w-12 h-12 border-4 border-slate-200 border-t-primario rounded-full animate-spin mb-4"></div>
          <p class="text-[10px] font-black uppercase tracking-widest mt-2">Sincronizando información...</p>
        </div>

        <form v-else @submit.prevent="handleUpdate" class="flex-1 flex flex-col space-y-10">
          
          <div>
            <label class="block text-[10px] font-black text-slate-400 uppercase mb-4 ml-2 tracking-widest">Actualizar Usuario (Login)</label>
            <div class="relative">
              <UserCircleIcon class="w-6 h-6 absolute left-6 top-1/2 -translate-y-1/2 text-slate-300" />
              <input
                v-model="profile.username"
                type="text"
                placeholder="Ej: juan_perez"
                class="w-full pl-16 pr-6 py-5 bg-slate-50 border border-slate-200 rounded-[2rem] focus:ring-4 focus:ring-primario/10 focus:border-primario focus:bg-white outline-none transition-all duration-300 font-bold text-slate-700 text-sm"
              />
            </div>
          </div>

          <!-- Contraseña -->
          <div class="bg-slate-50 p-8 md:p-10 rounded-[2.5rem] border border-slate-100 mt-6 lg:mt-12">
            <h4 class="text-[10px] sm:text-xs font-black text-slate-500 uppercase tracking-widest mb-8 flex items-center gap-3">
              <KeyIcon class="w-5 h-5" /> Cambio de Contraseña (Opcional)
            </h4>
            
            <div class="grid grid-cols-1 md:grid-cols-2 gap-8">
              <div>
                <label class="block text-[9px] font-black text-slate-400 uppercase mb-3 ml-2 tracking-widest">Nueva Contraseña</label>
                <div class="relative">
                  <input
                    v-model="passwordData.newPassword"
                    :type="showNew ? 'text' : 'password'"
                    placeholder="••••••••"
                    class="w-full px-6 py-5 bg-white border border-slate-200 rounded-3xl focus:ring-4 focus:ring-primario/10 outline-none transition-all duration-300 font-bold text-slate-700 text-sm"
                  />
                  <button type="button" @click="showNew = !showNew" class="absolute right-6 top-5 text-slate-400 hover:text-primario transition-colors">
                    <EyeIcon v-if="!showNew" class="w-6 h-6" />
                    <EyeSlashIcon v-else class="w-6 h-6" />
                  </button>
                </div>
              </div>

              <div>
                <label class="block text-[9px] font-black text-slate-400 uppercase mb-3 ml-2 tracking-widest">Confirmar Contraseña</label>
                <div class="relative">
                  <input
                    v-model="passwordData.confirmPassword"
                    :type="showConfirm ? 'text' : 'password'"
                    placeholder="••••••••"
                    class="w-full px-6 py-5 bg-white border border-slate-200 rounded-3xl focus:ring-4 focus:ring-primario/10 outline-none transition-all duration-300 font-bold text-slate-700 text-sm"
                  />
                  <button type="button" @click="showConfirm = !showConfirm" class="absolute right-6 top-5 text-slate-400 hover:text-primario transition-colors">
                    <EyeIcon v-if="!showConfirm" class="w-6 h-6" />
                    <EyeSlashIcon v-else class="w-6 h-6" />
                  </button>
                </div>
              </div>
            </div>
            
            <p v-if="passwordData.newPassword && passwordData.newPassword !== passwordData.confirmPassword" class="text-[9px] font-black text-red-500 mt-6 uppercase tracking-widest bg-red-50 p-3 rounded-xl border border-red-100 text-center">
              Las contraseñas no coinciden
            </p>
          </div>

          <div class="mt-auto pt-10 flex justify-end items-center">
            <button
              type="submit"
              :disabled="isUpdating || (passwordData.newPassword && passwordData.newPassword !== passwordData.confirmPassword)"
              class="w-full sm:w-auto bg-primario text-white px-14 py-6 rounded-[2rem] font-black text-xs uppercase tracking-[0.2em] shadow-xl shadow-primario/30 hover:bg-indigo-900 transition-all hover:-translate-y-1 active:scale-95 disabled:opacity-50 disabled:cursor-not-allowed disabled:-translate-y-0 disabled:shadow-none flex justify-center items-center gap-3"
            >
               <CheckCircleIcon v-if="!isUpdating" class="w-6 h-6" />
               <ArrowPathIcon v-else class="w-6 h-6 animate-spin" />
              {{ isUpdating ? 'Guardando...' : 'Guardar' }}
            </button>
          </div>

        </form>
      </section>
    </div>
  </div>
</template>

<script setup>
import { reactive, ref, onMounted } from 'vue';
import { UserIcon, BuildingOfficeIcon, EnvelopeIcon, CalendarDaysIcon, ShieldCheckIcon, UserCircleIcon, KeyIcon, EyeIcon, EyeSlashIcon, CheckCircleIcon, ArrowPathIcon, ExclamationTriangleIcon } from '@heroicons/vue/24/outline'
import { useAuthStore } from '../stores/auth'; // ✅ Usar el mismo store que el login
import clienteAxios from '../api/axios';        // ✅ Usar el mismo axios que le login

const auth = useAuthStore();
const loading = ref(true);
const isUpdating = ref(false);
const errorMsg = ref('');

const showNew     = ref(false)
const showConfirm = ref(false)

const profile = reactive({
  username: '',
  nombre_user: '',
  email: '',
});

const passwordData = reactive({
  newPassword: '',
  confirmPassword: ''
});

const fetchUserData = async () => {
  try {
    loading.value = true;
    errorMsg.value = '';

    // ✅ El ID viene del store (lo guardó el login: user_id: '4' en el token)
    // ✅ Lee del store, y si no, directo de localStorage
const userId = auth.user_id || localStorage.getItem('user_id'); // Asegúrate que tu store expone esto
    
    if (!userId) {
      errorMsg.value = "No se encontró el ID del usuario en sesión.";
      loading.value = false;
      return;
    }

    // ✅ URL correcta según tu Django: /api/usuarios/{id}/
    const response = await clienteAxios.get(`usuarios/${userId}/`);

    profile.username    = response.data.username    || '';
    profile.nombre_user = response.data.nombre_user || '';
    profile.email       = response.data.email       || '';

  } catch (error) {
    console.error("Error al cargar perfil:", error);
    if (error.response) {
      errorMsg.value = `Error ${error.response.status}: ${JSON.stringify(error.response.data)}`;
    } else if (error.request) {
      errorMsg.value = "Sin respuesta del servidor. ¿Está corriendo Django?";
    } else {
      errorMsg.value = "Error: " + error.message;
    }
  } finally {
    loading.value = false;
  }
};

onMounted(() => {
  fetchUserData();
});

const handleUpdate = async () => {
  if (passwordData.newPassword && passwordData.newPassword !== passwordData.confirmPassword) {
    alert("Las contraseñas no coinciden");
    return;
  }

  try {
    isUpdating.value = true;

    // ✅ Lee del store, y si no, directo de localStorage
    const userId = auth.user_id || localStorage.getItem('user_id');

    const payload = {
      username: profile.username,
    };

    if (passwordData.newPassword) {
      payload.password = passwordData.newPassword;
    }

    // ✅ URL correcta: /api/usuarios/{id}/
    await clienteAxios.patch(`usuarios/${userId}/`, payload);

    alert("¡Perfil actualizado con éxito!");
    passwordData.newPassword    = '';
    passwordData.confirmPassword = '';

  } catch (error) {
    const mensajeError = error.response?.data?.detail || JSON.stringify(error.response?.data) || "Error al conectar";
    alert("Error al actualizar: " + mensajeError);
  } finally {
    isUpdating.value = false;
  }
};
</script>
