<template>
  <div class="p-4 md:p-10 space-y-8">
    <div class="max-w-7xl mx-auto">

      <div class="mb-10 flex flex-col md:flex-row md:items-end justify-between gap-6">
        <div>
          <span class="text-[10px] font-black text-rosa-comp uppercase tracking-[0.2em]">Control de Activos</span>
          <h1 class="text-4xl font-black text-slate-800 tracking-tighter uppercase">Inventario de Unidades</h1>
          <p class="text-slate-500 text-sm font-medium mt-1">
            Sede: <span class="text-primario font-bold">{{ auth.nombre_deposito || 'Acceso Global' }}</span>
          </p>
        </div>

        <router-link v-if="auth.rol === 'OPERADOR'" to="/registrar" custom v-slot="{ navigate }">
          <button @click="navigate" class="flex items-center justify-center bg-primario text-white px-8 py-4 rounded-[1.5rem] hover:scale-105 transition-all shadow-lg shadow-primario/20 font-black text-[10px] uppercase tracking-widest gap-2">
            <PlusIcon class="w-5 h-5" /> Registrar Entrada
          </button>
        </router-link>
      </div>

      <div class="flex flex-col md:flex-row gap-4 mb-8 bg-white p-4 rounded-[2rem] border border-slate-100 shadow-sm">
        <div class="relative flex-1">
          <MagnifyingGlassIcon class="w-5 h-5 absolute left-5 top-1/2 -translate-y-1/2 text-slate-400" />
          <input v-model="search" type="text" placeholder="Buscar por placa, marca o modelo..." class="w-full bg-slate-50 border-none rounded-2xl pl-14 pr-6 py-4 text-sm font-bold text-slate-700 outline-none focus:ring-2 focus:ring-primario/20 transition-all" />
        </div>

        <div v-if="auth.rol === 'SUPER'" class="flex items-center gap-3 px-4">
          <span class="text-[10px] font-black text-slate-400 uppercase">Sede:</span>
          <select v-model="filtroSede" class="bg-slate-50 border-none rounded-xl px-4 py-3 text-[10px] font-black uppercase outline-none focus:ring-2 focus:ring-primario/20">
            <option value="TODAS">TODAS</option>
            <option v-for="s in sedes" :key="s.id" :value="s.id">{{ s.nombre }}</option>
          </select>
        </div>
      </div>

      <div class="hidden md:block bg-white border border-slate-100 rounded-[2.5rem] shadow-sm overflow-hidden">
        <table class="min-w-full divide-y divide-slate-100">
          <thead class="bg-slate-50/50">
            <tr>
              <th class="px-8 py-5 text-left text-[10px] font-black text-slate-400 uppercase tracking-widest">Vehículo</th>
              <th class="px-8 py-5 text-left text-[10px] font-black text-slate-400 uppercase tracking-widest">Año/Color</th>
              <th class="px-8 py-5 text-left text-[10px] font-black text-slate-400 uppercase tracking-widest">Placas</th>
              <th class="px-8 py-5 text-left text-[10px] font-black text-slate-400 uppercase tracking-widest">Estatus</th>
              <th class="px-8 py-5 text-center text-[10px] font-black text-slate-400 uppercase tracking-widest">Ver</th>
            </tr>
          </thead>
          <tbody class="divide-y divide-slate-50">
            <tr v-for="ingreso in filteredVehiculos" :key="ingreso.id">
              <td class="px-8 py-6">
                <div class="flex flex-col">
                  <span class="text-sm font-black text-slate-800">{{ ingreso.vehiculo.marca }}</span>
                  <span class="text-[11px] text-slate-400 font-bold uppercase">{{ ingreso.vehiculo.modelo }}</span>
                </div>
              </td>
              <td class="px-8 py-6 text-xs font-bold text-slate-600">
                {{ ingreso.vehiculo.anio }} | <span class="text-slate-400 italic">{{ ingreso.vehiculo.color_actual }}</span>
              </td>
              <td class="px-8 py-6">
                <span class="bg-slate-800 text-white px-3 py-1.5 rounded-xl text-[10px] font-mono font-black uppercase">
                  {{ ingreso.vehiculo.placas }}
                </span>
              </td>
             
              <td class="px-8 py-6 text-center">
                <button @click="verDetalles(ingreso.id)"> <EyeIcon class="w-5 h-5" />
                </button>
              </td>
            </tr>
          </tbody>
        </table>
      </div>

      <div class="md:hidden space-y-4">
        <div v-for="v in filteredVehiculos" :key="v.id" class="bg-white p-6 rounded-[2rem] border border-slate-100 shadow-sm relative overflow-hidden">
          <div :class="v.estatus_actual === 'DENTRO' ? 'bg-green-500' : 'bg-blue-500'" class="absolute left-0 top-0 bottom-0 w-1.5"></div>
          
          <div class="flex justify-between items-start mb-4">
            <div>
              <h3 class="font-black text-slate-800 text-sm uppercase">{{ v.marca }} {{ v.modelo }}</h3>
              <span class="bg-slate-100 text-slate-600 px-2 py-1 rounded-lg text-[10px] font-mono font-bold">{{ v.placas }}</span>
            </div>
            <button @click="verDetalles(v.id)" class="p-3 bg-slate-50 rounded-2xl text-primario">
              <EyeIcon class="w-5 h-5" />
            </button>
          </div>

          <div class="grid grid-cols-2 gap-4 border-t border-slate-50 pt-4">
            <div>
              <p class="text-[9px] font-black text-slate-300 uppercase">Año/Color</p>
              <p class="text-[10px] font-bold text-slate-600">{{ v.anio }} | {{ v.color_actual }}</p>
            </div>
            <div class="text-right">
              <p class="text-[9px] font-black text-slate-300 uppercase">Estatus</p>
              <span :class="claseEstatus(v.estatus_actual)" class="text-[9px] font-black uppercase">{{ v.estatus_actual }}</span>
            </div>
          </div>
        </div>
      </div>

    </div>
  </div>
</template>
<script setup>
import { ref, computed, onMounted } from 'vue'
import { useRouter } from 'vue-router'
import { useAuthStore } from '../stores/auth'
import clienteAxios from '../api/axios'
import { PlusIcon, EyeIcon, MagnifyingGlassIcon } from '@heroicons/vue/24/outline'

const auth = useAuthStore()
const router = useRouter()
const search = ref('')
const ingresos = ref([]) // Cambiamos 'vehiculos' por 'ingresos'
const sedes = ref([])
const filtroSede = ref('TODAS')

// 1. Cargar ingresos desde la API
const cargarIngresos = async () => {
  try {
    // IMPORTANTE: Si eres ADMIN/OPE, el backend ya filtra por tu id_deposito
    // pero si eres SUPER, puedes ver todos o filtrar.
    let url = 'ingresos/'
    
    const res = await clienteAxios.get(url)
    ingresos.value = res.data
  } catch (e) {
    console.error("Error cargando inventario:", e)
  }
}

const cargarSedes = async () => {
  if (auth.rol !== 'SUPER') return
  try {
    const res = await clienteAxios.get('depositos/')
    sedes.value = res.data
  } catch (e) { console.error(e) }
}

// 2. Lógica de filtrado inteligente (Backend + Frontend)
const filteredVehiculos = computed(() => {
  return ingresos.value.filter(ing => {
    const v = ing.vehiculo
    
    // 1. Filtro de búsqueda (Texto)
    const matchesSearch = 
      v.marca.toLowerCase().includes(search.value.toLowerCase()) ||
      v.modelo.toLowerCase().includes(search.value.toLowerCase()) ||
      v.placas.toLowerCase().includes(search.value.toLowerCase()) ||
      ing.folio.toLowerCase().includes(search.value.toLowerCase())

    // 2. Filtro de Sede (Solo aplica si eres SUPER)
    let matchesSede = true
    if (auth.rol === 'SUPER') {
      // Usamos == (doble igual) para evitar líos entre String y Number
      matchesSede = filtroSede.value === 'TODAS' || ing.deposito == filtroSede.value
    }

    return matchesSearch && matchesSede
  })
})

const verDetalles = (id) => {
  // Pasamos el ID del INGRESO, no del vehículo, 
  // porque los detalles (inspección, daños) dependen del ingreso.
  localStorage.setItem('ingreso_detalle_id', id)
  router.push('/Detallesauto')
}

const claseEstatus = (estatus) => {
  // Ahora el estatus viene del vehículo dentro del ingreso
  return estatus === 'DENTRO' 
    ? 'bg-green-50 text-green-600 border-green-100' 
    : 'bg-blue-50 text-blue-600 border-blue-100'
}

onMounted(() => {
  cargarIngresos()
  cargarSedes()
})
</script>
